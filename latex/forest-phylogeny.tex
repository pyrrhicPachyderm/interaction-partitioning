%A forest style for drawing phylogenetic trees.
\usepackage[edges]{forest}

\forestset{
	phylogeny/.style={
		forked edges,
		delay={where content={}{shape=coordinate}{}}, %Make empty nodes leave no gaps.
		before typesetting nodes={
			where n children=0{tier=terminus}{},
		},
		for tree={
			grow=east,
			align=left,
			anchor=west,
			s sep=0, %Reduce the separation between the leaf nodes.
		}
	}
}
